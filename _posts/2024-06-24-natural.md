# natural.c

#include <stdio.h>

int main(void)

{
    int n;

    // Get the number of natural numbers to print

    printf("Enter the number of natural numbers to print: ");

    scanf("%d", &n);


    // Loop to print the first n natural numbers

    for (int i = 1; i <= n; i++)

    {

        printf("%d ", i);

    }

    printf("\n"); // Print a newline at the end

    return 0;
    
}
