### max

#include <stdio.h>

#include <cs50.h>

int main(void)

{

    int num1, num2, max;

    // Get the numbers from the user

    printf("Enter the first number: ");

    scanf("%d", &num1);


    printf("Enter the second number: ");

    scanf("%d", &num2);


    // Determine the maximum number

    if (num1 > num2)

    {

        max = num1;

    }

    else

    {

        max = num2;

    }

    // Print the maximum number

    printf("The maximum of %d and %d is %d\n", num1, num2, max);


    return 0;
    
}
